// Enhanced partner tracking system for clean URLs class PartnerTracker { constructor() { this.basePartnerURL = "https://7shining.ch/home"; this.init(); } init() { // Check if we're on a clean partner URL this.detectCleanPartnerURL(); // Set up package button handlers this.setupPackageButtons(); } // Convert decimal partner ID to hex toHex(decimal) { return parseInt(decimal).toString(16).toUpperCase(); } // Convert hex back to decimal partner ID toDecimal(hex) { return parseInt(hex, 16); } // Detect clean partner URLs like /packages/A1B2C3 detectCleanPartnerURL() { const path = window.location.pathname; const patterns = [ /\/packages\/([A-F0-9]+)$/i, /\/join\/([A-F0-9]+)$/i, /\/gold\/([A-F0-9]+)$/i, /\/([A-F0-9]{4,})$/i // Direct hex at end of URL ]; for (let pattern of patterns) { const match = path.match(pattern); if (match) { const hexCode = match[1]; const partnerID = this.toDecimal(hexCode); // Store partner ID sessionStorage.setItem('partnerID', partnerID); localStorage.setItem('partnerID', partnerID); console.log(`Clean URL detected: ${hexCode} ‚Üí Partner ID: ${partnerID}`); return partnerID; } } // Fallback: check for traditional ?partnerid= parameter const urlParams = new URLSearchParams(window.location.search); const traditionalPartnerID = urlParams.get('partnerid'); if (traditionalPartnerID) { sessionStorage.setItem('partnerID', traditionalPartnerID); localStorage.setItem('partnerID', traditionalPartnerID); return traditionalPartnerID; } return null; } // Get stored partner ID getPartnerID() { return sessionStorage.getItem('partnerID') || localStorage.getItem('partnerID') || null; } // Setup package button click handlers setupPackageButtons() { document.addEventListener('DOMContentLoaded', () => { // Find all package buttons const packageButtons = document.querySelectorAll('a[href*="/addproduct/"]'); packageButtons.forEach(button => { button.addEventListener('click', (e) => { const partnerID = this.getPartnerID(); if (partnerID) { e.preventDefault(); const originalURL = button.href; const newURL = `${originalURL}?partnerid=${partnerID}`; console.log(`Redirecting with partner ID: ${partnerID}`); console.log(`Original URL: ${originalURL}`); console.log(`New URL: ${newURL}`); window.location.href = newURL; } }); }); }); } // Generate clean partner URL generateCleanURL(partnerID, page = 'packages') { const hexCode = this.toHex(partnerID); return `https://7shining.ch/${page}/${hexCode}`; } } // Initialize the partner tracker const partnerTracker = new PartnerTracker();
